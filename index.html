Code: Simple Bridge


from dataclasses import dataclass

@dataclass
class Bridge:
    start: float
    deltas: list
    def result(self) -> float:
        return self.start + sum(self.deltas)

print( Bridge(18.4, [+0.6, +0.4, -0.3]).result() )
